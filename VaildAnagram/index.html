class Solution:

# 思路：数组排序后比较差异
#    def isAnagram(self, s: str, t: str) -> bool:
#        return sorted(s) == sorted(t)

# 思路：手动模拟hashtable，将字符串”a-z“的ASCII码作key，
#计数求差异
    def isAnagram(self, s: str, t: str) -> bool:
        arr1, arr2 = [0]*26, [0]*26
        for i in s:
            arr1[ord(i) - ord('a')] += 1
        for i in t:
            arr2[ord(i) - ord('a')] += 1
        return arr1 == arr2
